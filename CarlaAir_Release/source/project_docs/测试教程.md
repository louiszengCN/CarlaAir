## 一、快速开始

### 1.1 一键启动

```bash
cd /mnt/data1/tianle/carla_source
./carlaAir.sh
```

脚本会自动：
1. 启动 UE4Editor（Town10HD 地图）
2. 等待 CARLA (端口 2000) 和 AirSim (端口 41451) 就绪
3. 打印 "Ready!" 提示

首次启动（含车辆/行人资源 DDC 构建）约需 10 分钟，之后约 1 分钟。

### 1.2 激活 Python 环境

```bash
conda activate simworld
```

### 1.3 验证连接

```bash
# 验证 CARLA
python3 -c "import carla; c=carla.Client('localhost',2000); print('CARLA:', c.get_world().get_map().name)"

# 验证 AirSim
python3 -c "import airsim; c=airsim.MultirotorClient(port=41451); c.confirmConnection()"
```

### 1.4 停止仿真器

```bash
./carlaAir.sh --kill
```

### 2.2 天气控制示例

```python
import carla

client = carla.Client('localhost', 2000)
world = client.get_world()

# 预设天气
world.set_weather(carla.WeatherParameters.ClearNoon)        # 晴天中午
world.set_weather(carla.WeatherParameters.HardRainNoon)      # 暴雨中午
world.set_weather(carla.WeatherParameters.ClearSunset)       # 晴天日落
world.set_weather(carla.WeatherParameters.WetCloudyNoon)     # 阴天潮湿

# 自定义天气
weather = carla.WeatherParameters(
    cloudiness=50.0,          # 云量 (0-100)
    precipitation=30.0,       # 降水 (0-100)
    sun_altitude_angle=70.0,  # 太阳高度角
    fog_density=10.0,         # 雾浓度
    wetness=50.0,             # 地面湿度
)
world.set_weather(weather)

# 读取当前天气
w = world.get_weather()
print(f"太阳高度: {w.sun_altitude_angle}, 降水: {w.precipitation}")
```

### 2.3 生成车辆

```python
import carla
import random

client = carla.Client('localhost', 2000)
world = client.get_world()
bp_lib = world.get_blueprint_library()

# 查看可用车辆
vehicle_bps = bp_lib.filter('vehicle.*')
print(f"可用车辆类型: {len(vehicle_bps)}")
for bp in vehicle_bps:
    print(f"  {bp.id}")

# 获取生成点
spawn_points = world.get_map().get_spawn_points()
print(f"生成点数量: {len(spawn_points)}")

# 生成一辆 Tesla Model3
bp = bp_lib.find('vehicle.tesla.model3')
vehicle = world.spawn_actor(bp, spawn_points[0])
print(f"已生成: {vehicle.type_id}, ID: {vehicle.id}")

# 开启自动驾驶
vehicle.set_autopilot(True)

# 生成多辆随机车辆
vehicles = []
for i in range(10):
    bp = random.choice(vehicle_bps)
    try:
        v = world.spawn_actor(bp, spawn_points[i + 1])
        v.set_autopilot(True)
        vehicles.append(v)
    except Exception as e:
        print(f"生成失败: {e}")
print(f"成功生成 {len(vehicles)} 辆车")

# 清理
import time
time.sleep(10)  # 观察10秒
for v in vehicles:
    v.destroy()
vehicle.destroy()
```

### 2.4 生成行人

```python
import carla
import random

client = carla.Client('localhost', 2000)
world = client.get_world()
bp_lib = world.get_blueprint_library()

# 查看可用行人
walker_bps = bp_lib.filter('walker.pedestrian.*')
print(f"可用行人类型: {len(walker_bps)}")

# 生成行人
spawn_points = world.get_map().get_spawn_points()
bp = random.choice(walker_bps)
sp = spawn_points[20]
sp.location.z += 1.0  # 稍微抬高，避免地面碰撞
walker = world.spawn_actor(bp, sp)
print(f"已生成行人: {walker.type_id}")

# 生成行人控制器并让行人行走
walker_controller_bp = bp_lib.find('controller.ai.walker')
controller = world.spawn_actor(walker_controller_bp, carla.Transform(), walker)
controller.start()
# 设置目标点
target = spawn_points[25].location
controller.go_to_location(target)
controller.set_max_speed(1.4)  # 步行速度 m/s
print(f"行人正在走向 ({target.x:.0f}, {target.y:.0f})")

import time
time.sleep(15)

# 清理
controller.stop()
controller.destroy()
walker.destroy()
```

### 2.5 传感器挂载

```python
import carla
import time

client = carla.Client('localhost', 2000)
world = client.get_world()
bp_lib = world.get_blueprint_library()
spawn_points = world.get_map().get_spawn_points()

# 生成车辆
vehicle = world.spawn_actor(bp_lib.find('vehicle.tesla.model3'), spawn_points[0])
vehicle.set_autopilot(True)

# 挂载 RGB 相机
cam_bp = bp_lib.find('sensor.camera.rgb')
cam_bp.set_attribute('image_size_x', '1280')
cam_bp.set_attribute('image_size_y', '720')
cam_bp.set_attribute('fov', '90')
cam_transform = carla.Transform(carla.Location(x=1.5, z=2.0))
camera = world.spawn_actor(cam_bp, cam_transform, attach_to=vehicle)

# 挂载 LiDAR
lidar_bp = bp_lib.find('sensor.lidar.ray_cast')
lidar_bp.set_attribute('range', '50')
lidar_bp.set_attribute('channels', '32')
lidar_bp.set_attribute('points_per_second', '100000')
lidar_transform = carla.Transform(carla.Location(z=2.5))
lidar = world.spawn_actor(lidar_bp, lidar_transform, attach_to=vehicle)

# 接收数据
def save_image(image):
    image.save_to_disk(f'/tmp/carla_cam/{image.frame}.png')

def save_lidar(data):
    data.save_to_disk(f'/tmp/carla_lidar/{data.frame}.ply')

import os
os.makedirs('/tmp/carla_cam', exist_ok=True)
os.makedirs('/tmp/carla_lidar', exist_ok=True)

camera.listen(save_image)
lidar.listen(save_lidar)

print("传感器正在采集数据... (10秒)")
time.sleep(10)

# 清理
camera.stop()
lidar.stop()
camera.destroy()
lidar.destroy()
vehicle.destroy()
print("完成。图像: /tmp/carla_cam/  点云: /tmp/carla_lidar/")
```

### 3.2 基本飞行控制

```python
import airsim

client = airsim.MultirotorClient(port=41451)
client.confirmConnection()

# 启用 API 控制并解锁
client.enableApiControl(True)
client.armDisarm(True)

# 起飞
client.takeoffAsync().join()
print("已起飞")

# 飞到指定高度（NED 坐标：z=-20 表示地面以上 20m）
client.moveToZAsync(-20, 5).join()   # z=-20m, 速度 5m/s
print("已升至 20m")

# 飞到指定位置
client.moveToPositionAsync(50, 30, -20, 10).join()  # x=50, y=30, z=-20, 速度 10m/s
print("已飞到 (50, 30)")

# 速度控制（持续飞行 3 秒）
client.moveByVelocityAsync(5, 0, 0, 3).join()  # vx=5m/s 向前飞 3 秒

# 悬停
client.hoverAsync().join()

# 降落
client.landAsync().join()
client.armDisarm(False)
client.enableApiControl(False)
print("已降落")
```

### 3.3 相机图像采集

```python
import airsim
import os

client = airsim.MultirotorClient(port=41451)
client.confirmConnection()
client.enableApiControl(True)
client.armDisarm(True)
client.takeoffAsync().join()
client.moveToZAsync(-15, 3).join()

os.makedirs('/tmp/drone_images', exist_ok=True)

# RGB 图像
responses = client.simGetImages([
    airsim.ImageRequest("0", airsim.ImageType.Scene, False, False)
])
airsim.write_file('/tmp/drone_images/rgb.png', responses[0].image_data_uint8)
print(f"RGB: {responses[0].width}x{responses[0].height}")

# 深度图
responses = client.simGetImages([
    airsim.ImageRequest("0", airsim.ImageType.DepthVis, False, False)
])
airsim.write_file('/tmp/drone_images/depth.png', responses[0].image_data_uint8)
print(f"深度图: {responses[0].width}x{responses[0].height}")

# 语义分割
responses = client.simGetImages([
    airsim.ImageRequest("0", airsim.ImageType.Segmentation, False, False)
])
airsim.write_file('/tmp/drone_images/seg.png', responses[0].image_data_uint8)
print(f"语义分割: {responses[0].width}x{responses[0].height}")

# 一次获取多种图像
responses = client.simGetImages([
    airsim.ImageRequest("0", airsim.ImageType.Scene, False, False),
    airsim.ImageRequest("0", airsim.ImageType.DepthPlanar, True),  # float depth
    airsim.ImageRequest("0", airsim.ImageType.Segmentation, False, False),
])
print(f"批量获取: {len(responses)} 张图像")

client.landAsync().join()
client.armDisarm(False)
client.enableApiControl(False)
print("图像保存至 /tmp/drone_images/")
```

### 3.4 航线飞行（Waypoint Mission）

```python
import airsim

client = airsim.MultirotorClient(port=41451)
client.confirmConnection()
client.enableApiControl(True)
client.armDisarm(True)
client.takeoffAsync().join()

# 定义航线（NED 坐标）
waypoints = [
    (0, 0, -15),     # 起点上空 15m
    (30, 0, -15),    # 前进 30m
    (30, 30, -20),   # 右转 + 升高
    (0, 30, -20),    # 返回
    (0, 0, -15),     # 回到起点上空
]

speed = 8  # m/s

for i, (x, y, z) in enumerate(waypoints):
    print(f"飞向航点 {i}: ({x}, {y}, {z})")
    client.moveToPositionAsync(x, y, z, speed).join()
    pos = client.getMultirotorState().kinematics_estimated.position
    print(f"  到达: ({pos.x_val:.1f}, {pos.y_val:.1f}, {pos.z_val:.1f})")

client.landAsync().join()
client.armDisarm(False)
client.enableApiControl(False)
print("航线飞行完成")
```

### 4.2 自定义联合场景示例

```python
import carla
import airsim
import time

# 连接
carla_client = carla.Client('localhost', 2000)
carla_client.set_timeout(10.0)
world = carla_client.get_world()
bp_lib = world.get_blueprint_library()

airsim_client = airsim.MultirotorClient(port=41451)
airsim_client.confirmConnection()

# 场景 1：暴雨中的交通 + 无人机巡逻
world.set_weather(carla.WeatherParameters.HardRainNoon)

# 生成车流
spawn_points = world.get_map().get_spawn_points()
vehicles = []
for i in range(15):
    bp = bp_lib.filter('vehicle.*')[i % len(bp_lib.filter('vehicle.*'))]
    try:
        v = world.spawn_actor(bp, spawn_points[i])
        v.set_autopilot(True)
        vehicles.append(v)
    except:
        pass

# 无人机起飞巡逻
airsim_client.enableApiControl(True)
airsim_client.armDisarm(True)
airsim_client.takeoffAsync().join()
airsim_client.moveToZAsync(-25, 5).join()

# 沿路线巡逻
for x in range(0, 100, 20):
    airsim_client.moveToPositionAsync(x, 0, -25, 10).join()
    # 采集数据
    responses = airsim_client.simGetImages([
        airsim.ImageRequest("0", airsim.ImageType.Scene, False, False)
    ])
    print(f"位置 x={x}: 图像 {responses[0].width}x{responses[0].height}")

# 清理
airsim_client.landAsync().join()
airsim_client.armDisarm(False)
airsim_client.enableApiControl(False)
for v in vehicles:
    v.destroy()
```

## 五、遥测数据与状态查询

### 5.1 AirSim 无人机遥测

```python
import airsim

client = airsim.MultirotorClient(port=41451)
client.confirmConnection()

state = client.getMultirotorState()

# 位置 (NED 坐标)
pos = state.kinematics_estimated.position
print(f"位置: x={pos.x_val:.2f}, y={pos.y_val:.2f}, z={pos.z_val:.2f}")

# 速度
vel = state.kinematics_estimated.linear_velocity
print(f"速度: vx={vel.x_val:.2f}, vy={vel.y_val:.2f}, vz={vel.z_val:.2f}")

# 姿态（四元数）
ori = state.kinematics_estimated.orientation
print(f"姿态: w={ori.w_val:.3f}, x={ori.x_val:.3f}, y={ori.y_val:.3f}, z={ori.z_val:.3f}")

# GPS
gps = client.getGpsData()
print(f"GPS: {gps}")

# 降落状态 (0=飞行中, 1=已降落)
print(f"降落状态: {state.landed_state}")

# IMU
imu = client.getImuData()
print(f"IMU 角速度: ({imu.angular_velocity.x_val:.2f}, {imu.angular_velocity.y_val:.2f}, {imu.angular_velocity.z_val:.2f})")
```

## 六、常见问题排查

| 问题 | 解决方案 |
|------|----------|
| `Connection refused` (CARLA) | 确认 UE4 已启动，检查 `ss -tlnp \| grep 2000` |
| `Connection refused` (AirSim) | 确认 AirSim API 就绪，检查 `ss -tlnp \| grep 41451` |
| 天气设置无效 | 确认使用的是 SimWorldGameMode（检查日志 `Game class is 'SimWorldGameMode'`） |
| 车辆/行人生成失败 | 检查生成点是否被占用，尝试其他 spawn_points 索引 |
| 无人机不动 | 确认已调用 `enableApiControl(True)` 和 `armDisarm(True)` |
| 画面卡顿 | 降低分辨率或使用 `--quality=Low`，减少同时生成的 actor 数量 |
| 长时间运行崩溃 | 确保启动参数包含 `-TexturePoolSize=2048` |
| 首次启动很慢 | DDC 和着色器需首次编译，耐心等待，后续启动会很快 |
| 键盘控制无响应 | 确保终端窗口有焦点；安装 `pip install pynput` |
| `Version mismatch` 警告 | 可忽略，CARLA 客户端库版本与源码编译版本略有差异但兼容 |
| 查看实时日志 | `./carlaAir.sh --log` 或 `tail -f /tmp/carla_simworld.log` |
